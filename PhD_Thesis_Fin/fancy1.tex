% This file contains fancy layout stuff

% Fancy headers, configurable page headers and footers
% see http://www.ctan.org/pkg/fancyhdr
\usepackage{fancyhdr}

 %This section puts the chapter and the page number on the top of the pages
\pagestyle{fancy}
\renewcommand{\chaptermark}[1]{\markboth{#1}{}}
\renewcommand{\sectionmark}[1]{\markright{\thesection\ #1}}
\fancyhf{}
\fancyhead[LE,RO]{\bfseries\thepage}
\fancyhead[LO]{\bfseries\rightmark}
\fancyhead[RE]{\bfseries\leftmark}
\renewcommand{\headrulewidth}{0.5pt}
\renewcommand{\footrulewidth}{0pt}
\addtolength{\headheight}{0.5pt}
\setlength{\footskip}{0in}
\renewcommand{\footruleskip}{0pt}
\fancypagestyle{plain}{%
  \fancyhead{}
  \renewcommand{\headrulewidth}{0pt}
}

\makeatletter
\def\cleardoublepage{\clearpage\if@twoside \ifodd\c@page\else
    \hbox{}
    \thispagestyle{plain}
    \newpage
    \if@twocolumn\hbox{}\newpage\fi\fi\fi}
\makeatother \clearpage{\pagestyle{plain}\cleardoublepage}

% Change chapter style 
% CHAPTER X - BLABLA
\titleformat{\chapter}[display]
  % Shape
  {\LARGE % \Large \huge \Huge - Font size
   \filcenter % \filleft \filright - alignment
   \bf} % boldface
   % CHAPTER X 
  {\titlerule[1pt]\vspace{1pt}\titlerule % double line over 
  \vspace{1pc} % Space between lines and text
   \MakeUppercase{\chaptertitlename} % CHAPTER
   \thechapter} % X
   % Vertical distance between chapter X and 
  {20pt}
   % Shape for BLABLA
  {\Huge % \Large \huge \Huge - Font size
   \filcenter % \filleft \filright - alignment
   \bf\sc} % boldface, smallcaps
  % Stuff that comes after BLABLABAL
  [\vspace{2ex} \titlerule] % Append a line under the title
